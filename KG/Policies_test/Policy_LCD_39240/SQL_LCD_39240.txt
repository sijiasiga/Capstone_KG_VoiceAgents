```sql
SELECT *
FROM patients
WHERE
  -- Restriction 1: Epidural steroid injection (ESI) requires diagnostic support for lumbar radiculopathy.
  -- Restriction 2: Epidural steroid injection (ESI) requires diagnostic support for cervical radiculopathy.
  -- Restriction 3: Epidural steroid injection (ESI) requires diagnostic support for thoracic radiculopathy.
  -- Restriction 4: Epidural steroid injection (ESI) requires diagnostic support for radicular pain.
  -- Restriction 5: Epidural steroid injection (ESI) requires diagnostic support for neurogenic claudication due to disc herniation.
  -- Restriction 6: Epidural steroid injection (ESI) requires diagnostic support for neurogenic claudication due to osteophyte or osteophyte complexes.
  -- Restriction 7: Epidural steroid injection (ESI) requires diagnostic support for neurogenic claudication due to severe degenerative disc disease producing foraminal or central spinal stenosis.
  -- Restriction 8: Epidural steroid injection (ESI) requires diagnostic support for post-laminectomy syndrome.
  -- Restriction 9: Epidural steroid injection (ESI) requires diagnostic support for acute herpes zoster associated pain.
  (diagnosis_radiculopathy_lumbar = TRUE
  OR
  diagnosis_radiculopathy_cervical = TRUE
  OR
  diagnosis_radiculopathy_thoracic = TRUE
  OR
  diagnosis_radicular_pain = TRUE
  OR
  (diagnosis_neurogenic_claudication = TRUE AND diagnosis_disc_herniation = TRUE)
  OR
  (diagnosis_neurogenic_claudication = TRUE AND diagnosis_osteophyte = TRUE)
  OR
  (diagnosis_neurogenic_claudication = TRUE AND diagnosis_degenerative_disc_disease_severe = TRUE AND (diagnosis_spinal_stenosis_foraminal = TRUE OR diagnosis_spinal_stenosis_central = TRUE))
  OR
  diagnosis_post_laminectomy_syndrome = TRUE
  OR
  diagnosis_acute_herpes_zoster_pain = TRUE)
  AND
  -- Restriction 10: Radiculopathy, radicular pain, or neurogenic claudication must be severe enough to greatly impact quality of life or function, with objective pain and functional assessments performed at baseline and follow-up.
  (diagnosis_radiculopathy_lumbar = TRUE OR diagnosis_radiculopathy_cervical = TRUE OR diagnosis_radiculopathy_thoracic = TRUE OR diagnosis_radicular_pain = TRUE OR diagnosis_neurogenic_claudication = TRUE) AND pain_scale_baseline = TRUE AND functional_assessment_baseline = TRUE AND pain_scale_followup = TRUE AND functional_assessment_followup = TRUE
  AND
  -- Restriction 11: Patient must have pain duration of at least four weeks and either failed noninvasive conservative care or be unable to tolerate it, or have acute herpes zoster refractory to conservative management without a four-week wait.
  (pain_duration_weeks >= 4 AND (conservative_care_failure = TRUE OR noninvasive_care_intolerance = TRUE)) OR acute_herpes_zoster_refractory = TRUE
  AND
  -- Restriction 12: Epidural steroid injections must be performed under CT or fluoroscopy image guidance with contrast.
  (image_guidance_ct = TRUE OR image_guidance_fluoroscopy = TRUE) AND contrast_agent_used = TRUE
  AND
  -- Restriction 13: If a patient has a documented allergy to low molecular weight nonionic contrast or is pregnant, ultrasound guidance without contrast may be considered for ESI.
  (allergy_contrast_media = TRUE OR patient_pregnancy_status = TRUE) AND image_guidance_ultrasound = TRUE AND contrast_agent_used = FALSE
  AND
  -- Restriction 14: Transforaminal epidural steroid injections (TFESIs) are limited to a maximum of two levels in one spinal region during a single session.
  tfesi_spinal_levels_max <= 2 AND tfesi_spinal_region_count_max <= 1
  AND
  -- Restriction 15: Caudal epidural steroid injections (CESIs) and interlaminar epidural steroid injections (ILESIs) are limited to a maximum of one level during a single session.
  cesi_spinal_levels_max <= 1 AND ilesi_spinal_levels_max <= 1
  AND
  -- Restriction 16: Transforaminal epidural steroid injections (TFESIs) may be performed bilaterally only when clinically indicated.
  tfesi_bilateral = TRUE
  AND
  -- Restriction 17: Repeat ESI is medically reasonable and necessary if the first injection provided at least 50% sustained improvement in pain relief and/or function for at least three months.
  (repeat_esi_pain_relief_min_percent >= 50 OR repeat_esi_function_improvement_min_percent >= 50) AND repeat_esi_improvement_duration_months >= 3
  AND
  -- Restriction 18: If an initial ESI fails to provide sufficient response, a repeat ESI can be performed after 14 days, potentially using a different approach, level, or medication, with documented rationale.
  conservative_care_failure = TRUE AND repeat_esi_initial_failure_window_days >= 14
  AND
  -- Restriction 19: An initial injection of contrast is required to confirm epidural placement, unless contraindication to contrast exists.
  initial_esi_contrast_required = TRUE
  AND
  -- Restriction 20: Subsequent epidural steroid injections should include corticosteroids and may be combined with anesthetics or saline.
  subsequent_esi_corticosteroid_component = TRUE AND (subsequent_esi_anesthetic_component = TRUE OR subsequent_esi_saline_component = TRUE)
  AND
  -- Restriction 21: Epidural steroid injections (ESIs) must be performed in conjunction with conservative treatments.
  esi_conjunction_conservative_treatment = TRUE
  AND
  -- Restriction 22: Patient must be part of an active rehabilitation program, home exercise program, or functional restoration program.
  patient_rehabilitation_program = TRUE
  AND
  -- Restriction 23: Injections performed without image guidance are not covered.
  image_guidance_required = TRUE
  AND
  -- Restriction 24: Ultrasound guidance for ESI is only considered in cases of documented contraindication to contrast media (e.g., allergy) or pregnancy.
  (allergy_contrast_media = TRUE OR patient_pregnancy_status = TRUE) AND image_guidance_ultrasound = TRUE
  AND
  -- Restriction 25: ESIs performed with biologicals or other substances not FDA designated for this use are considered investigational and not medically reasonable and necessary.
  esi_biologicals_not_covered = TRUE AND esi_unapproved_substances_not_covered = TRUE
  AND
  -- Restriction 26: Performing multiple blocks (e.g., sympathetic, facet, trigger point injections) during the same session as ESIs is generally not covered.
  multiple_blocks_same_session_not_covered = TRUE
  AND
  -- Restriction 27: An ESI and a facet synovial cyst procedure may be performed in the same session as an exception to the multiple blocks rule.
  facet_synovial_cyst_esi_same_session_allowed = TRUE
  AND
  -- Restriction 28: Moderate sedation is not considered medically reasonable and necessary for ESI procedures.
  sedation_moderate_not_covered = TRUE
  AND
  -- Restriction 29: Deep sedation is not considered medically reasonable and necessary for ESI procedures.
  sedation_deep_not_covered = TRUE
  AND
  -- Restriction 30: General anesthesia is not considered medically reasonable and necessary for ESI procedures.
  anesthesia_general_not_covered = TRUE
  AND
  -- Restriction 31: Monitored Anesthesia Care (MAC) is not considered medically reasonable and necessary for ESI procedures.
  mac_anesthesia_not_covered = TRUE
  AND
  -- Restriction 32: Oral anxiolytics are typically sufficient for patients with needle phobia or anxiety during ESI procedures.
  oral_anxiolytics_sufficient = TRUE
  AND
  -- Restriction 33: ESIs for non-specific low back pain (LBP) are not medically reasonable and necessary.
  esi_for_nonspecific_LBP_not_covered = TRUE
  AND
  -- Restriction 34: ESIs for axial spine pain are not medically reasonable and necessary.
  esi_for_axial_spine_pain_not_covered = TRUE
  AND
  -- Restriction 35: ESIs for complex regional pain syndrome are not medically reasonable and necessary.
  esi_for_complex_regional_pain_syndrome_not_covered = TRUE
  AND
  -- Restriction 36: ESIs for widespread diffuse pain are not medically reasonable and necessary.
  esi_for_widespread_diffuse_pain_not_covered = TRUE
  AND
  -- Restriction 37: ESIs for pain from neuropathy from other causes are not medically reasonable and necessary.
  esi_for_neuropathy_other_causes_not_covered = TRUE
  AND
  -- Restriction 38: ESIs for cervicogenic headaches are not medically reasonable and necessary.
  esi_for_cervicogenic_headaches_not_covered = TRUE
  AND
  -- Restriction 39: ESIs are limited to a maximum of four sessions per spinal region in a rolling twelve-month period.
  esi_sessions_max_per_region_12months <= 4
  AND
  -- Restriction 40: Injecting more than one spinal region in the same session is not considered medically reasonable and necessary.
  esi_multiple_spinal_regions_same_session_not_covered = TRUE
  AND
  -- Restriction 41: Performing Caudal or Interlaminar Epidural Steroid Injections (CESIs/ILESIs) bilaterally is not medically reasonable and necessary.
  cesi_ilesi_bilateral_not_covered = TRUE
  AND
  -- Restriction 42: Prescribing a predetermined series of ESIs is not medically reasonable and necessary.
  predetermined_esi_series_not_covered = TRUE
  AND
  -- Restriction 43: Triamcinolone dose per ESI session should not exceed 80 mg.
  steroid_dose_triamcinolone_max_mg <= 80
  AND
  -- Restriction 44: Betamethasone dose per ESI session should not exceed 12 mg.
  steroid_dose_betamethasone_max_mg <= 12
  AND
  -- Restriction 45: Dexamethasone dose per ESI session should not exceed 15 mg.
  steroid_dose_dexamethasone_max_mg <= 15
  AND
  -- Restriction 46: Methylprednisolone dose per ESI session should not exceed 40 mg (as per WIP Benelux Workgroup recommendation).
  steroid_dose_methylprednisolone_max_mg <= 40
  AND
  -- Restriction 47: Treatment with ESI generally should not extend beyond 12 months.
  esi_treatment_duration_max_months <= 12
  AND
  -- Restriction 48: Frequent continuation of epidural steroid injections over 12 months may trigger a focused medical review.
  esi_continuation_beyond_12months_review_trigger = TRUE
  AND
  -- Restriction 49: For ESI continuation beyond 12 months, pain must be severe enough to cause significant functional or vocational disability.
  (esi_continuation_beyond_12months_functional_disability = TRUE OR esi_continuation_beyond_12months_vocational_disability = TRUE)
  AND
  -- Restriction 50: For ESI continuation beyond 12 months, ESI must provide at least 50% sustained improvement of pain and/or 50% objective improvement in function.
  (esi_continuation_beyond_12months_pain_relief_min_percent >= 50 OR esi_continuation_beyond_12months_function_improvement_min_percent >= 50)
  AND
  -- Restriction 51: For ESI continuation beyond 12 months, rationale for continuation must be documented.
  esi_continuation_beyond_12months_rationale_documented = TRUE
  AND
  -- Restriction 52: For ESI continuation beyond 12 months, the primary care provider must be notified.
  esi_continuation_beyond_12months_pcp_notification = TRUE
  AND
  -- Restriction 53: Use of ESIs beyond one year must be combined with multimodal therapy.
  esi_continuation_beyond_12months_multimodal_therapy_required = TRUE
  AND
  -- Restriction 54: ESIs are contraindicated for suspected or active localized spinal infection.
  contraindication_spinal_infection = TRUE
  AND
  -- Restriction 55: ESIs are contraindicated for significant systemic infection.
  contraindication_systemic_infection = TRUE
  AND
  -- Restriction 56: ESIs are contraindicated for compressive lesions of the spinal cord.
  contraindication_spinal_cord_compression = TRUE
  AND
  -- Restriction 57: ESIs are contraindicated for compressive lesions of the conus medullaris.
  contraindication_conus_medullaris_compression = TRUE
  AND
  -- Restriction 58: ESIs are contraindicated for compressive lesions of the cauda equina.
  contraindication_cauda_equina_compression = TRUE
  AND
  -- Restriction 59: ESIs are contraindicated for suspicion or major risk factors for cancer.
  contraindication_cancer_risk = TRUE
  AND
  -- Restriction 60: ESIs are absolutely contraindicated for infectious spondylitis.
  contraindication_infectious_spondylitis = TRUE
  AND
  -- Restriction 61: ESIs are absolutely contraindicated for myelopathy.
  contraindication_myelopathy = TRUE
  AND
  -- Restriction 62: ESIs are absolutely contraindicated for inability to obtain informed consent.
  contraindication_inability_to_obtain_informed_consent = TRUE
  AND
  -- Restriction 63: ESIs are absolutely contraindicated for infection at the skin puncture site.
  contraindication_skin_puncture_site_infection = TRUE
  AND
  -- Restriction 64: Uncorrected anticoagulation therapy is a relative contraindication for ESI.
  relative_contraindication_uncorrected_anticoagulation = TRUE
  AND
  -- Restriction 65: Hypersensitivity to administered agents is a relative contraindication for ESI.
  relative_contraindication_hypersensitivity_agents = TRUE
  AND
  -- Restriction 66: Hepatitis is a relative contraindication for ESI.
  relative_contraindication_hepatitis = TRUE
  AND
  -- Restriction 67: Uncontrolled diabetes mellitus is a relative contraindication for ESI.
  relative_contraindication_uncontrolled_diabetes_mellitus = TRUE
  AND
  -- Restriction 68: Congestive heart failure is a relative contraindication for ESI.
  relative_contraindication_congestive_heart_failure = TRUE
  AND
  -- Restriction 69: Immunosuppressed state is a relative contraindication for ESI.
  relative_contraindication_immunosuppressed_state = TRUE
  AND
  -- Restriction 70: Patient improving on medical and physical therapy is a relative contraindication for ESI.
  relative_contraindication_patient_improving_medical_pt = TRUE
  AND
  -- Restriction 71: Severe spinal canal stenosis is a relative contraindication for ESI.
  relative_contraindication_severe_spinal_canal_stenosis = TRUE
  AND
  -- Restriction 72: No response to a previous well-performed ESI is a relative contraindication.
  relative_contraindication_no_response_previous_esi = TRUE
  AND
  -- Restriction 73: Complication to steroid therapy (e.g., Cushingâ€™s) is a relative contraindication for ESI.
  relative_contraindication_steroid_therapy_complication = TRUE
  AND
  -- Restriction 74: Minimizing corticosteroid exposure is recommended for high-risk individuals with a prior history of osteoporosis or osteopenia when considering ESI.
  contraindication_osteoporosis_history = TRUE
  AND
  -- Restriction 75: Interlaminar ESIs (ILESIs) should not be performed above C7.
  cervical_ilesi_max_level = 'C7'
  AND
  -- Restriction 76: Non-particulate steroids are recommended for cervical and thoracic spine ESIs.
  non_particulate_steroid_use_cervical_thoracic_spine_recommended_nass = TRUE
  AND
  -- Restriction 77: A maximum of two Transforaminal ESI procedures can be performed at a single setting.
  tfesi_sessions_max_same_setting_nass <= 2
  AND
  -- Restriction 78: A maximum of one Caudal ESI or Interlaminar ESI procedure can be performed at a single setting.
  cesi_ilesi_sessions_max_same_setting_nass <= 1
  AND
  -- Restriction 79: Caudal ESI or Interlaminar ESI should not be performed in conjunction with a Transforaminal ESI injection.
  cesi_ilesi_not_in_conjunction_with_tfesi_nass = TRUE
  AND
  -- Restriction 80: Most patients respond to 3 or less ESI injections for a specific episode of back and radicular pain.
  successful_esi_injections_count_max <= 3
  AND
  -- Restriction 81: Performing more than 3 ESI injections requires specific non-evidence-based criteria such as surgical planning, new injuries, inaccurate prior injections, exacerbation, or high-risk surgical candidates.
  esi_beyond_3_injections_criteria_nass = TRUE
  AND
  -- Restriction 82: Epidural steroid injections (ESIs) are suggested for very short-term reduction (up to two weeks) of radicular low back pain.
  esi_for_radicular_LBP_short_term_benefit_window_weeks <= 2
  AND
  -- Restriction 83: Triamcinolone is not recommended for ESI use due to an FDA label change.
  steroid_type_triamcinolone_not_for_esi = TRUE
  AND
  -- Restriction 84: Particulate steroids are not recommended for cervical TFESIs due to safety concerns.
  steroid_type_particulate_not_recommended_cervical_tfesi = TRUE
  AND
  -- Restriction 85: Non-particulate steroids are recommended for cervical TFESIs.
  steroid_type_non_particulate_recommended_cervical_tfesi = TRUE
  AND
  -- Restriction 86: Non-particulate steroids are recommended for lumbar TFESIs due to improved safety profile.
  steroid_type_non_particulate_recommended_lumbar_tfesi = TRUE
  AND
  -- Restriction 87: A period of up to 14 days may be needed to assess the clinical response after an ESI.
  esi_assess_clinical_response_window_days <= 14
  AND
  -- Restriction 88: Systemic effects on the hypothalamic-pituitary-adrenal (HPA) axis may last three weeks or longer after an ESI.
  hpa_axis_systemic_effects_duration_weeks >= 3;
```
