-- Policy: NCD - Diagnosis and Treatment of Impotence
-- Total restrictions: 5
SELECT *
FROM patients
WHERE
  -- Restriction 1: Patient must have a diagnosis of impotence
  diagnosis_impotence = TRUE
  AND
  -- Restriction 2: Treatment involves surgical implantation of a penile prosthesis for impotence
  procedure_penile_prosthesis_implantation = TRUE
  AND
  -- Restriction 3: Treatment involves medical methods for impotence
  treatment_medical_impotence = TRUE
  AND
  -- Restriction 4: Treatment involves psychotherapeutic methods for impotence
  treatment_psychotherapeutic_impotence = TRUE
  AND
  -- Restriction 5: Psychiatric service limitations apply if treatment for impotence is furnished to non-hospital inpatients in connection with a mental condition
  (patient_hospital_inpatient_flag = FALSE AND mental_condition_present_flag = TRUE);
