```sql
SELECT *
FROM patients
WHERE
  -- Policy restrictions: 34
  -- Restriction 1: SCS therapy is indicated for the relief of chronic intractable pain.
  diagnosis_chronic_intractable_pain = TRUE
  AND
  -- Restriction 2: SCS is best suited for neuropathic pain.
  -- Restriction 3: SCS may have limited value in nociceptive severe, intractable pain.
  (
    scs_therapy_pain_type_neuropathic = TRUE
    OR
    scs_therapy_pain_type_nociceptive = TRUE
  )
  AND
  -- Restriction 4: Performance and documentation of an effective SCS trial is a prerequisite for permanent nerve stimulation.
  scs_trial_prerequisite_for_permanent_implant = TRUE
  AND
  -- Restriction 5: A new SCS trial is not needed for stimulator replacement due to battery change.
  -- Restriction 6: A new SCS trial is not needed for stimulator replacement due to malfunction.
  -- Restriction 7: A new SCS trial is not needed for stimulator replacement due to end of stimulator life.
  (
    scs_replacement_no_trial_needed_reason = 'battery change'
    OR
    scs_replacement_no_trial_needed_reason = 'malfunction'
    OR
    scs_replacement_no_trial_needed_reason = 'end of stimulator life'
  )
  AND
  -- Restriction 8: SCS therapy must be considered a late option after more conservative therapies have been tried.
  prior_conservative_therapy_tried = TRUE
  AND
  -- Restriction 9: Patient must undergo careful screening, evaluation and diagnosis by a multidisciplinary team prior to implantation.
  patient_multidisciplinary_screening_required = TRUE
  AND
  -- Restriction 10: Psychological evaluation is a required component of patient screening.
  patient_psychological_evaluation_required = TRUE
  AND
  -- Restriction 11: Physical evaluation is a required component of patient screening.
  patient_physical_evaluation_required = TRUE
  AND
  -- Restriction 12: Patient must not have active substance abuse issues.
  patient_active_substance_abuse_exclusion = FALSE
  AND
  -- Restriction 13: Patient must undergo proper education, discussion, and disclosure of risks and benefits.
  patient_education_disclosure_required = TRUE
  AND
  -- Restriction 14: Temporary neurostimulator trials should preferably be placed in an ASC or outpatient hospital setting.
  scs_trial_setting_preferred IN ('ASC', 'outpatient hospital setting')
  AND
  -- Restriction 15: Temporary neurostimulator trials can be done in an office setting if all sterility, equipment, professional training, and support personnel are available.
  scs_trial_setting_office_criteria = 'all sterility, equipment, professional training, and support personnel required for the proper surgery, and follow up of the patient are available'
  AND
  -- Restriction 16: Permanent neurostimulators must be placed in an ASC or hospital.
  scs_permanent_implant_setting_required IN ('ASC', 'hospital')
  AND
  -- Restriction 17: Physicians performing SCS trials in the office setting must have like privileges at a local hospital or ASC.
  -- Restriction 18: Physicians performing SCS trials in the office setting must be subspecialty boarded in Pain Medicine by the American Board of Anesthesiology.
  (
    physician_scs_trial_office_qualifications = 'privileges at a local hospital or ASC'
    OR
    physician_scs_trial_office_qualifications = 'subspecialty boarded in Pain Medicine by the American Board of Anesthesiology'
  )
  AND
  -- Restriction 19: Only patients who experience a positive response to an SCS trial should proceed to permanent implantation.
  scs_trial_positive_response_for_permanent_implant = TRUE
  AND
  -- Restriction 20: A successful SCS trial should be associated with at least a 50% reduction of target pain.
  -- Restriction 21: A successful SCS trial should be associated with at least a 50% reduction of analgesic medications.
  (
    scs_trial_success_pain_reduction_threshold_percent >= 50.0
    OR
    scs_trial_success_analgesic_medication_reduction_threshold_percent >= 50.0
  )
  AND
  -- Restriction 22: A successful SCS trial must show some element of functional improvement.
  scs_trial_success_functional_improvement_required = TRUE
  AND
  -- Restriction 23: For patients with reflex sympathetic dystrophy, lower levels of improvement and longer trial periods may be expected.
  diagnosis_reflex_sympathetic_dystrophy_scs_trial_impact = 'may show lower levels of improvement and takes longer periods for improvement'
  AND
  -- Restriction 24: Physicians with a trial to permanent implant ratio less than 50% will be subject to post-payment review.
  physician_trial_to_implant_ratio_threshold_percent < 50.0
  AND
  -- Restriction 25: Noridian will reimburse for a maximum of 2 SCS leads.
  scs_leads_max_count <= 2
  AND
  -- Restriction 26: Noridian will reimburse for a maximum of 16 SCS contacts.
  scs_contacts_max_count <= 16
  AND
  -- Restriction 27: Noridian will reimburse for a maximum of 2 SCS trials per anatomic spinal region per patient per lifetime.
  scs_trials_max_count_per_region_lifetime <= 2
  AND
  -- Restriction 28: Exceptions to the maximum number of SCS trials are allowed for technical limitations for initial trials.
  -- Restriction 29: Exceptions to the maximum number of SCS trials are allowed for use of different modalities of stimulation, including new technology.
  (
    scs_trials_max_count_exception_criteria = 'technical limitations for initial trials'
    OR
    scs_trials_max_count_exception_criteria = 'use of different modalities of stimulation, including new technology'
  )
  AND
  -- Restriction 30: A repeat SCS trial is not appropriate after initial failure unless there are extenuating circumstances.
  scs_repeat_trial_allowed_criteria = 'extenuating circumstances that lead to trial failure'
  AND
  -- Restriction 31: Diagnosis code for encounter for adjustment and management of neurostimulator.
  -- Restriction 32: Diagnosis codes for specific spinal disorders.
  -- Restriction 33: Diagnosis codes including causalgia and complications of internal prosthetic devices affecting lower limb or trunk.
  -- Restriction 34: Diagnosis codes including complications of internal prosthetic devices affecting upper limb.
  (
    diagnosis_code_ICD10 IN ('Z45.42')
    OR
    diagnosis_code_ICD10 IN ('M48.061', 'M48.062')
    OR
    diagnosis_code_ICD10 IN ('G57.73', 'T85.113A', 'T85.113D', 'T85.113S', 'T85.123A', 'T85.123D', 'T85.123S', 'T85.193A', 'T85.193D', 'T85.193S')
    OR
    diagnosis_code_ICD10 IN ('T85.112A', 'T85.112D', 'T85.112S', 'T85.122A', 'T85.122D', 'T85.122S', 'T85.192A', 'T85.192D', 'T85.192S')
  );
```
